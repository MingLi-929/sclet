Users can also use plotColData() to visualize the QC data.

# VlnPlot-like
p1 <- plotColData(sce, y = 'nCount_RNA')

# FeatureScatter-like
p2 <- plotColData(sce, x = "nCount_RNA", y = "percent.mt")
plot_list(p1, p2)
